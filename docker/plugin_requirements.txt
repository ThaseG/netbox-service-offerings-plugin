# Third-party NetBox plugins baked into the demo image (see docker/Dockerfile-Plugins).
# Pinned deliberately, same philosophy as versions.sh: nothing resolves to
# "latest" so an upstream plugin release can't silently break the instance.
# Bump versions in a PR after checking the plugin's compatibility table
# against the NETBOX_VERSION pinned in versions.sh.

# https://pypi.org/project/netboxlabs-netbox-custom-objects/ — compatible with NetBox 4.5.2-4.6.x
netboxlabs-netbox-custom-objects==0.6.0

# https://github.com/netbox-community/netbox-topology-views — declares support for NetBox 4.5.x;
# no hard version gate in the plugin code, so it loads against 4.6.x, but is untested there upstream.
netbox-topology-views==4.5.1

# https://github.com/DanSheps/netbox-lifecycle — requires NetBox 4.1+
netbox-lifecycle==1.1.9

# https://github.com/Onemind-Services-LLC/netbox-secrets — plugin 3.1.x is the NetBox 4.6.x line
netbox-secrets==3.1.0
